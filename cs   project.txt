void borderCreator()
{
	gotoxy(0, 0);
	for (int y = 0; y < 25; y++)
	{
		if (y == 0 || y == 24)
			for (int x = 1; x < 80; x++)
			{
				gotoxy(x, y);
				cout << "█";
			}
		else
		{
			gotoxy(1, y);
			cout << "█";
			gotoxy(79, y);
			cout << "█";
		}
	}
	gotoxy(2, 2);

}

}
void delay(int x)
{
	float j = 0;

	for (int i = 1; i < x; i++);
	   for (int k=1;k<100;k++)
       {
           j=43*21/37;
           k=34*19;
           k=k/10;
           j=j/435;
       }
}
